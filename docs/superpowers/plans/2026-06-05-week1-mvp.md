# Week 1 MVP Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use $superpower-subagents (recommended) or $superpower-executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking via update_plan.

**Goal:** Build the first runnable front-back MVP for the SMPL body modeling service: mock input -> backend -> OBJ/manifest output -> minimal 3D viewer.

**Architecture:** Start with a Python local-first backend package that exposes a CLI and stable data contracts. The backend writes `manifest.json` plus a simple OBJ mesh into `outputs/<task_id>/`; the frontend is a lightweight Vite + Three.js viewer that can load a sample manifest and mesh. Real SMPL/SMPL-X and dataset adapters remain replaceable backends after the mock pipeline is stable.

**Tech Stack:** Python 3, pytest, argparse/dataclasses/json/pathlib, Node.js, Vite, Three.js, vanilla TypeScript or JavaScript.

from __future__ import annotations

from dataclasses import asdict, dataclass, field
from typing import Any


@dataclass(frozen=True)
class BodyInput:
    source_type: str
    subject_id: str
    input_files: list[str]
    metadata: dict[str, Any] = field(default_factory=dict)

    def to_dict(self) -> dict[str, Any]:
        return asdict(self)


@dataclass(frozen=True)
class ModelResult:
    task_id: str
    model_type: str
    status: str
    output_paths: dict[str, str] = field(default_factory=dict)
    errors: list[str] = field(default_factory=list)

    def to_dict(self) -> dict[str, Any]:
        return asdict(self)


@dataclass(frozen=True)
class Manifest:
    task_id: str
    body_input_source: str
    result: ModelResult

    def to_dict(self) -> dict[str, Any]:
        return {
            "task_id": self.task_id,
            "body_input_source": self.body_input_source,
            "result": self.result.to_dict(),
        }
```

class MockInputAdapter:

        if not input_path.exists():
            raise FileNotFoundError(f"mock input file not found: {input_path}")

        payload = json.loads(input_path.read_text(encoding="utf-8"))
        subject_id = str(payload["subject_id"])
        return BodyInput(
            source_type="mock",
            subject_id=subject_id,
            input_files=[str(input_path)],
            metadata={key: value for key, value in payload.items() if key != "subject_id"},
        )

from __future__ import annotations

import json
from pathlib import Path
from typing import Any

from .contracts import Manifest


def write_obj(mesh: dict[str, Any], output_path: Path) -> None:
    lines = ["# mock body mesh"]
    for vertex in mesh["vertices"]:
        lines.append(f"v {vertex[0]} {vertex[1]} {vertex[2]}")
    for face in mesh["faces"]:
        lines.append(f"f {face[0]} {face[1]} {face[2]}")
    output_path.write_text("\n".join(lines) + "\n", encoding="utf-8")


def write_manifest(manifest: Manifest, output_path: Path) -> None:
    output_path.write_text(
        json.dumps(manifest.to_dict(), indent=2, ensure_ascii=False) + "\n",
        encoding="utf-8",
    )
```

- [ ] **Step 4: Implement pipeline**

Create `project/backend/src/smpl_service/pipeline.py`:

```python
from __future__ import annotations

from pathlib import Path

from .contracts import Manifest, ModelResult
from .input_adapters import MockInputAdapter
from .model_backends import MockBackend
from .output_writer import write_manifest, write_obj


def run_modeling_task(
    source_type: str,
    input_path: Path,
    model_type: str,
    output_dir: Path,
) -> Path:
    if source_type != "mock":
        raise ValueError(f"unsupported source_type for MVP: {source_type}")
    if model_type != "mock":
        raise ValueError(f"unsupported model_type for MVP: {model_type}")

    adapter = MockInputAdapter()
    body_input = adapter.load(input_path)
    task_id = f"job_{body_input.subject_id}"
    task_dir = output_dir / task_id
    task_dir.mkdir(parents=True, exist_ok=True)

    backend = MockBackend()
    mesh = backend.run(body_input)
    mesh_path = task_dir / "body.obj"
    manifest_path = task_dir / "manifest.json"

    write_obj(mesh, mesh_path)
    result = ModelResult(
        task_id=task_id,
        model_type=model_type,
        status="success",
        output_paths={"mesh": str(mesh_path), "manifest": str(manifest_path)},
        errors=[],
    )
    manifest = Manifest(task_id=task_id, body_input_source=body_input.source_type, result=result)
    write_manifest(manifest, manifest_path)
    return manifest_path
```

async function loadManifest() {
  const response = await fetch('/sample/manifest.json');
  if (!response.ok) {
    throw new Error(`failed to load manifest: ${response.status}`);
  }
  return response.json();
}

function loadObj(meshPath) {
  const loader = new OBJLoader();
  loader.load(
    meshPath,
    (object) => {
      object.traverse((child) => {
        if (child.isMesh) {
          child.material = new THREE.MeshStandardMaterial({
            color: 0x7c9cbf,
            roughness: 0.65,
            metalness: 0.05,
          });
        }
      });
      scene.add(object);
      statusEl.textContent = 'Loaded';
    },
    undefined,
    (error) => {
      statusEl.textContent = 'Mesh load failed';
      console.error(error);
    },
  );
}

function animate() {
  controls.update();
  renderer.render(scene, camera);
  requestAnimationFrame(animate);
}

loadManifest()
  .then((manifest) => {

    loadObj(result.output_paths.mesh);
  })
  .catch((error) => {
    statusEl.textContent = 'Manifest load failed';

animate();
